pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Build Images') {
            steps {
                dir('src/docker') {
                    sh 'docker compose build'
                }
            }
        }
        stage('Debug List') {
            steps {
                    sh 'ls -l /var/lib/jenkins/workspace/IEV-JankisPipelineMkI'
                    sh 'ls -l /var/lib/jenkins/workspace/IEV-JankisPipelineMkI/src'
                 }
        }
        stage('Start Services') {
            steps {
                dir('src/docker') {
                    sh 'docker compose up -d'
                }
            }
        }
        stage('Test') {
            steps {
                echo 'No tests implemented yet'
            }
        }
        stage('Shutdown') {
            steps {
                dir('src/docker') {
                    sh 'docker compose down'
                }
            }
        }
    }
    post {
        always {
            dir('src/docker') {
                sh 'docker compose down || true'
            }
        }
    }
}
