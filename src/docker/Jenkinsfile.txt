pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build Images') {
            steps {
                dir('src/docker') {
                    sh 'docker compose build'
                }
            }
        }
        stage('Start Services') {
            steps {
                dir('src/docker') {
                    sh 'docker compose up -d'
                }
            }
        }
        stage('Test') {
            steps {
                // Tests, API-Aufrufe etc.
            }
        }
        stage('Shutdown') {
            steps {
                dir('src/docker') {
                    sh 'docker compose down'
                }
            }
        }
    }
    post {
        always {
            dir('src/docker') {
                sh 'docker compose down || true'
            }
        }
    }
}
